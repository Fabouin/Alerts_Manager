alertmanager_toggle_invitations = {
	scope = character
	is_shown = {
		NOT = {
			has_character_flag = alertmanager_invitations_disabled
		}
	}
	effect = {
		if = {
			limit = {
				has_character_flag = alertmanager_invitations_disabled
			}
			remove_character_flag = alertmanager_invitations_disabled
		}
		else = {
			add_character_flag = alertmanager_invitations_disabled
		}
	}
}
